if(FIPS_EMSCRIPTEN OR FIPS_RASPBERRYPI OR FIPS_ANDROID)
    add_definitions(-DSOKOL_GLES2)
elseif(FIPS_WINDOWS)
    add_definitions(-DSOKOL_D3D11)
elseif(FIPS_OSX)
    add_definitions(-DSOKOL_METAL)
else()
    add_definitions(-DSOKOL_GLCORE33)
endif()

fips_begin_lib(sokol)
    fips_vs_warning_level(3)
    if (FIPS_OSX)
        fips_files(sokol.m)
        if (FIPS_IOS)
            fips_frameworks_osx(UIKit Metal MetalKit)
        else()
            fips_frameworks_osx(Cocoa QuartzCore Metal MetalKit)
        endif()
    else()
        fips_files(sokol.c)
    endif()
fips_end_lib()

fips_begin_app(z1013 windowed)
    fips_vs_warning_level(3)
    fips_files(z1013.c)
    fips_dir(roms)
    fips_generate(FROM z1013-roms.yml TYPE dump SOURCE z1013-roms.c HEADER z1013-roms.h)
    fips_deps(sokol)
fips_end_app()
